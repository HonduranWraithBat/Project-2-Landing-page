  <section id="ingredients" class="card">
    <h2 class="section-heading">Ingredients</h2>
    
    <ul class="ingredient-list">
      <li class="ingredient-item">1 lb shrimp</li>
      <li class="ingredient-item">3 garlic cloves</li>
      <li class="ingredient-item">2 tbsp olive oil</li>
      <li class="ingredient-item">1 lime</li>
      <li class="ingredient-item">Salt & pepper</li>
      <li class="ingredient-item">Cilantro</li>
    </ul>
    
    <img class="recipe-image"
         
         src="https://images.unsplash.com/photo-1604908177522-402c6d0f5c4b"
         alt="Shrimp and ingredients">
  </section>
  
  <section id="instructions" class="card">
    <h2 class="section-heading">Instructions</h2>
    
    <ol class="instruction-list">
      <li class="step-item">Heat oil in pan at medium heat</li>
      <li class="step-item">Cook minced garlic</li>
      <li class="step-item">Add shrimp</li>
      <li class="step-item">Add lime + seasoning</li>
      <li class="step-item">mix and cook for 12 minutes</li>
      <li class="step-item"> Garnish and serve</li>
    </ol>
